# keep one OOC = 1
data_OOC1 <- data_raw %>%
  filter(OCC == 1)

# Create a new data frame with the sum of DV for each individual, excluding time 0
y_dt <- data_OOC1 %>%
  filter(TIME != 0) %>%
  group_by(ID) %>%
  summarize(Y = sum(DV, na.rm = TRUE), .groups = "drop")

# keep time 0 data
time0 <- data_OOC1 %>%
  filter(TIME == 0) 

# Merge the two data frames to include the sum of DV and the time 0 data
d1 <- time0 %>%
  left_join(y_dt, by = "ID")

# Converts RACE and SEX to factor variables and keeps only these variables: Y,DOSE,AGE,SEX,RACE,WT,HT
d1 <- d1 %>%
  mutate(
    RACE = as.factor(RACE),
    SEX = as.factor(SEX),
    DOSE = as.factor(DOSE)
  ) %>% 
  select(Y, DOSE, AGE, SEX, RACE, WT, HT)

The logistic model using DOSE alone showed poor predictive performance, with a ROC-AUC of 0.592 and accuracy of 0.867, where the high accuracy is largely driven by class imbalance. Adding all predictors substantially improved the model, with ROC-AUC increasing to 0.978 and accuracy reaching 0.95. However, this does not mean sex can be clinically predicted from these variables. And since gender is something we already know, it has no practical significance.
